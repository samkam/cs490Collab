			return usurv_logs ? usurv_logs : new Array();
		},
		isQuarantined: function(){
			log('Log length: ' + this.log().length);
			if(this.log().length == 0) return false;
			
			// count number of skips
			var skips = 0;
			var timeFrame = 86400000 * this.skipsTimeFrame;
			for(var i=this.log().length-1; i>=0; i--){
				var elapsed = new Date().getTime() - parseInt(this.log()[i].date);
				if(this.log()[i].type == 'completed' || elapsed > timeFrame) break;
				skips++; 
				if(skips >= this.skipsBeforeQuarantine && elapsed < timeFrame) break;
			}
			log('Skips: ' + skips);
			log('Elapsed: ' + parseFloat(elapsed / 86400000).toFixed(5) + ' days');
			
			// figure out quarantine period
			var lastLog = this.log()[this.log().length-1];
			log('Last log: '+lastLog.type);
			if(skips >= this.skipsBeforeQuarantine) var qp = this.quarantinePeriods['skipLimit'];
			else var qp = this.quarantinePeriods[lastLog.type];
			
			// get difference in days
			var diff = new Date().getTime() - parseInt(lastLog.date); // now - last surveyed
			diff /= 86400000;
			log('Quarantine period: ' + qp + ' days');
			log('Time since last survey: ' + parseFloat(diff).toFixed(5) + ' days');
			
			// Are we still in quarantine period?
			if(diff >= qp) { 
				log('Not quarantined\n');
				return false;
			} else {
				log('Quarantined\n');
				return true;
			}
			
		},
		add: function(type, date){
			if(!date) date = new Date().getTime();
			log('Add: '+type);
			if(type == 'activeSkip' || type == 'completed') {
				this.removeLast();
				log('Last log removed');
			}
			this.log().push({'date': date , 'type': type});	
			this.updateCookie();
		},
		updateCookie: function(){
			if(this.log().length > this.maxLogLength) this.log().splice(this.log().length - this.maxLogLength, this.maxLogLength);
			setCookie('surveyLogs2', JSON.stringify(this.log()), 365, true);
		},
		removeLast: function(){
			this.log().pop();
			this.updateCookie();
		}
	};
