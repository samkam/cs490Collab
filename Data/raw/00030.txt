	function loadAltImage()
	{
		var div = "";
		var divClass="";
		var alink = "";
		var img = "";
		var allImg = Venda.ProductDetail.getAllImages();
		var allImglength = allImg['altthumb'].length;
		var i = 0;
		var current_alt_array = "";
		var current_alt_index_array = "";

			div = document.createElement("div");
			img = document.createElement("img");
			alink = document.createElement("a");
			divClass = "80006256_" + i;
			img.src = allImg['altthumb'][i];
			alink.href = "javascript:void(0);";

			jQuery(alink).append(img);

			jQuery(div).append(alink);
			jQuery(div).addClass(divClass);
			
			if(i == 0){ jQuery(div).addClass('first'); jQuery(div).addClass("selected"); }
			jQuery("#alt-view-wrapper .alt-view-image-list").append(div);
			count_alt_image++;
		}

		// If alt image more than 3, show button
		if(count_alt_image > 3){
			jQuery('#alt-view-wrapper .alt-bt-right .bt-alt-right').show(); 
			jQuery('#alt-view-wrapper .alt-bt-right .bt-alt-right').css('cursor','pointer');
		}
		// If alt image at less 3, hide right arrow

		jQuery("#alt-view-wrapper .alt-view-image-list div").click(function(){
			jQuery("#alt-view-wrapper .alt-view-image-list div").removeClass("selected");

			current_alt_array = this.className.split(" ");

			// if click on same div again

			{
				jQuery("." + current_alt).removeClass("selected");
				current_alt = current_alt_array[0];

				current_alt_index_array = current_alt.split("_");
				current_alt_index = current_alt_index_array[1];

				changeMainImage(current_alt_index);
			}
		});
		
	}

	function changeMainImage(alt_index)
	{	
		var img = "";
		var allImg = Venda.ProductDetail.getAllImages();

		// Remove old image
		if(jQuery(jqClassname + " img")){ jQuery(jqClassname + " img").remove(); }

		img = document.createElement("img");
		img.src = allImg['altmed'][alt_index];
		
		jQuery(jqClassname).attr("href", allImg['altzoom'][alt_index]);
		
		jQuery("#productdetail #alt-view-wrapper #imgWare").attr("href", allImg['altzoom'][alt_index]);

		jQuery(jqClassname).append(img);

		//Re-call jqZoom
		removeJzoom();
		activeJzoom();
	}
