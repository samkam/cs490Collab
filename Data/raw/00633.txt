        //if the height is below the maximum 160px so just show it all

            $('#moreDetailsButton').hide();

            //set the description to the smallest height
            $(descriptionContainer).height(visibleHeight);

            var newAvailableHeight = $('#productDetails').height();

            //if changing the description height didn't affect the size of the details container anyway, 
            //then set it back to the full height
            //(e.g. products with large images such as Keyboards or Guitars)
            if (newAvailableHeight == availableHeight && newAvailableHeight > 0 && availableHeight > 0) {
                visibleHeight = fullHeight;
                $('#moreDetailsButton').hide();
            } else {
                //try to fill as much whitespace as possible                
                if (descTop != undefined && descTop > 0) {

                    //with the description minimized, check if there is any more space for us to fit in...
                    //(e.g. items with extra images, or large items with large descriptions)

                    //resize if the overall product details reduced in height when the description changed

                        newAvailableHeight = newAvailableHeight - 30;
                        var newHeight = newAvailableHeight - descTop;
